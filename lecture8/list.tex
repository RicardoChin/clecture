{
  TNode *p = head;
  int i = 0;
  while(p != NULL)
  {
      printf("%3d\n", p->a);
      p = p->next;
      i++;
  }
  return i;

\begin{lstlisting}[xleftmargin=0.05\linewidth, linewidth=0.9\linewidth]
void deleteNode(int val, TNode *head)
{
   TNode *p = head, *q = head;
   while(p != NUL && p->a != val)
   {
       q = p;
       p = p->next;
   }
   if(p != NULL && p->a == val)
   {
       q->next = p->next;
       p->next = NULL;
       free(p);
   }
}
\end{lstlisting}
